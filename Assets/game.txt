// ==UserScript==
// @name         Starve.io AdBlocker (Key Activation)
// @namespace    https://starve.io/
// @version      1.2
// @description  Removes ads from Starve.io when you press 'X' in the main menu.
// @author       YourName
// @match        https://starve.io/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    let adsRemoved = false; // Track if ads have already been removed

    // Function to safely remove ad elements
    const removeAds = () => {
        const adSelectors = [
            'iframe[src*="ads"]', // Ad-related iframes
            'div[class*="ad"]', // Divs with "ad" in their class name
            'div[id*="ad"]', // Divs with "ad" in their ID
            '.ad-banner', // Common banner ads
            '#ad-container', // Example ad container
            '.ad-box', // Example ad box
            '.pop-up-ad', // Example pop-up ad
        ];

        adSelectors.forEach(selector => {
            const elements = document.querySelectorAll(selector);
            elements.forEach(el => {
                el.style.display = 'none'; // Hide instead of removing to avoid breaking layouts
                console.log(`Ad element hidden: ${selector}`);
            });
        });

        adsRemoved = true; // Mark ads as removed
        console.log('Ads have been removed!');
    };

    // Add a keyboard listener to detect 'X' key press
    document.addEventListener('keydown', (event) => {
        if (event.key === 'x' || event.key === 'X') {
            if (!adsRemoved) {
                console.log("Key 'X' pressed. Removing ads...");
                removeAds();
            } else {
                console.log('Ads have already been removed.');
            }
        }
    });

    console.log('Starve.io AdBlocker Script Loaded! Press "X" in the main menu to activate.');
})();
